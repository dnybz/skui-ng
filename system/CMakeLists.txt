set(system_src
    environment.h environment.cpp environment_windows.cpp environment_unix.cpp
   )
source_group(system FILES ${system_src})

# Platform-specific files
if(NOT WIN32)
  exclude_from_build(
                     environment_windows.cpp
                    )
endif()
if(NOT UNIX)
  exclude_from_build(
                     environment_unix.cpp
                    )
  if(NOT APPLE)
  endif()
endif()


add_library(system ${system_src})
target_link_libraries(system PUBLIC core)
